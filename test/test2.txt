/* Testar se um inteiro positivo é primo. */
bool is_prime(int n) {
int d;
d = 2;
if (n == 1) // 1 não é primo
return false;
while (d <= n) {
if (n%d == 0)
return false;
else
d = d+1;
}
return true;
}

int main() {
int n;
n = scan_int();
if (is_prime(n))
print_str("prime");
else
print_str("not prime");
}
